CREATE TABLE VEHICLE(
NUM_PLATE VARCHAR2(10),
OWNER VARCHAR2(15),
MANUFACTURER VARCHAR2(15),
MODEL VARCHAR2(15),
PRIMARY KEY (NUM_PLATE));

CREATE TABLE TECHNICIAN(
TID NUMBER(3),
NAME VARCHAR2(15),
SKILLS VARCHAR2(15),
PRIMARY KEY (TID));


CREATE TABLE BAY(
BID NUMBER(3),
DESCRIPTION VARCHAR(15),
BAY_TYPE VARCHAR(15),
PRIMARY KEY (BID));

CREATE TABLE JOB(
JNO NUMBER(3),
VEHICLE_ID VARCHAR2(10),
INFO VARCHAR2(15),
BID NUMBER(3),
TID NUMBER(3),
DATE_IN DATE,
DELIVERY_DATE DATE,
PRIMARY KEY(JNO),
FOREIGN KEY (VEHICLE_ID) REFERENCES VEHICLE (NUM_PLATE),
FOREIGN KEY (BID) REFERENCES BAY (BID),
FOREIGN KEY (TID) REFERENCES TECHNICIAN (TID));


CREATE TABLE BILL(
BNO NUMBER(3),
JNO NUMBER(3),
AMOUNT NUMBER(5),
PRIMARY KEY(BNO),
FOREIGN KEY(JNO) REFERENCES JOB (JNO));
